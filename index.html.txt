function enterChat() {
  myName = nameInput.value.trim();

  db.ref(`rooms/${ROOM}/lastActive`).once("value", s => {
    const lastActive = s.val();
    const now = Date.now();

      db.ref(`rooms/${ROOM}/messages`).remove();
    }
    db.ref(`rooms/${ROOM}/lastActive`).set(now);
    db.ref(`rooms/${ROOM}/lastActive`).onDisconnect().set(Date.now());
  });

  // Listen for messages
  db.ref(`rooms/${ROOM}/messages`).on("value", snap => {
    const val = snap.val();
    const allMsgs = val ? Object.entries(val).map(([k,v]) => ({...v,_key:k})).sort((a,b)=>a.ts-b.ts) : [];
    
    // Clear existing messages (except notice)
    while (msgs.children.length > 1) msgs.removeChild(msgs.lastChild);

    } else {
      emptyState.remove();
      allMsgs.forEach(msg => renderMessage(msg));
    }
    msgs.scrollTop = msgs.scrollHeight;
  });

function renderMessage(msg) {

  const av = document.createElement("div");
  av.className = "mini-av";
  av.textContent = initials(msg.senderName);
  
  const content = document.createElement("div");
  
  if (msg.type === "text") {
    const bub = document.createElement("div");
    bub.className = "bub";
    bub.textContent = msg.text;
    content.appendChild(bub);
  } else {
    const bub = document.createElement("div");
    bub.className = "bub media";
    const img = document.createElement("img");
    img.src = msg.text;
    img.className = "msg-img";
    img.onclick = () => { lightboxImg.src = msg.text; lightbox.classList.add("open"); };
    bub.appendChild(img);
    content.appendChild(bub);
  }
  
  const time = document.createElement("div");
  time.className = "msg-time";
  time.textContent = msg.time;
  content.appendChild(time);
  
  row.appendChild(av);
  row.appendChild(content);
  msgs.appendChild(row);
}

async function searchGifs(q) {
  const url = q
    ? `https://tenor.googleapis.com/v2/search?q=${encodeURIComponent(q)}&key=${TENOR_KEY}&limit=12&media_filter=gif`
    : `https://tenor.googleapis.com/v2/featured?key=${TENOR_KEY}&limit=12&media_filter=gif`;
  try {
    const res = await fetch(url);
    const data = await res.json();
    gifGrid.innerHTML = "";
    (data.results || []).forEach(g => {
      const url = g?.media_formats?.tinygif?.url || g?.media_formats?.gif?.url;
      if (!url) return;
      const item = document.createElement("div");
      item.className = "gif-item";
      const img = document.createElement("img");
      img.src = url;
      img.loading = "lazy";
      item.appendChild(img);
      item.onclick = () => {
        sendMessage(url, "gif");
        gifPanel.classList.remove("open");
        gifGrid.innerHTML = "";
        gifSearch.value = "";
      };
      gifGrid.appendChild(item);
    });
  } catch(e) {}
}
